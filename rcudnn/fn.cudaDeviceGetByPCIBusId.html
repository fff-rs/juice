pub unsafe extern "C" fn cudaDeviceGetByPCIBusId(
    device: *mut i32, 
    pciBusId: *const i8
) -> cudaError

\brief Returns a handle to a compute device

Returns in \p *device a device ordinal given a PCI bus ID string.

\param device - Returned device ordinal

\param pciBusId - String in one of the following forms: [domain]:[bus]:[device].[function] [domain]:[bus]:[device] [bus]:[device].[function] where \p domain, \p bus, \p device, and \p function are all hexadecimal values

\return ::cudaSuccess, ::cudaErrorInvalidValue, ::cudaErrorInvalidDevice \notefnerr \note_init_rt \note_callback

\sa ::cudaDeviceGetPCIBusId, ::cuDeviceGetByPCIBusId
